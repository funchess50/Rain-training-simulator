            function updateProgress() {
                let globalTotal = 0;
                let globalCompleted = 0;

                const phases = [1, 2, 3, 4, 5, 6, 7];
                phases.forEach(p => {
                    const checkboxes = document.querySelectorAll(`.step-checkbox[data-phase="phase${p}"]`);
                    if(checkboxes.length === 0) return;
                    
                    let phaseCompleted = 0;
                    checkboxes.forEach(cb => {
                        globalTotal++;
                        if (cb.checked) {
                            phaseCompleted++;
                            globalCompleted++;
                            cb.closest('.step-card').classList.add('completed');
                        } else {
                            cb.closest('.step-card').classList.remove('completed');
                        }
                    });
                    
                    const pct = Math.round((phaseCompleted / checkboxes.length) * 100);
                    // Update sidebar badges for this phase
                    const badges = document.querySelectorAll(`.progress-badge[data-phase="phase${p}"]`);
                    badges.forEach(badge => {
                        badge.innerText = `${pct}%`;
                        if(pct === 100) {
                            badge.classList.add('bg-emerald-100', 'text-emerald-700');
                            badge.classList.remove('bg-slate-100', 'text-slate-500');
                        } else {
                            badge.classList.remove('bg-emerald-100', 'text-emerald-700');
                            badge.classList.add('bg-slate-100', 'text-slate-500');
                        }
                    });
                });

                const globalPct = globalTotal === 0 ? 0 : Math.round((globalCompleted / globalTotal) * 100);
                progressChart.data.datasets[0].data = [globalPct, 100 - globalPct];
                progressChart.update();
                
                const mobileText = document.getElementById('mobile-progress-text');
                if(mobileText) mobileText.innerText = globalPct;
            }

            // Init listeners for all checkboxes
            document.querySelectorAll('.step-checkbox').forEach(cb => { 
                cb.addEventListener('change', updateProgress); 
            });

            // Accordions
            const labButtons = document.querySelectorAll('.lab-btn');
            labButtons.forEach(btn => {
                btn.addEventListener('click', function() {
                    const content = this.nextElementSibling;
                    const icon = this.querySelector('span:last-child');
                    content.classList.toggle('open');
                    if (content.classList.contains('open')) {
                        icon.style.transform = 'rotate(-180deg)';
                    } else {
                        icon.style.transform = 'rotate(0deg)';
                    }
                });
            });

            // Copy Buttons
            const copyButtons = document.querySelectorAll('.copy-btn');
            copyButtons.forEach(btn => {
                btn.addEventListener('click', function() {
                    const tempTextArea = document.createElement('textarea');
                    tempTextArea.innerHTML = this.getAttribute('data-copy');
                    navigator.clipboard.writeText(tempTextArea.value).then(() => {
                        const originalText = this.innerText;
                        this.innerText = 'Copied!';
                        this.classList.add('copied');
                        setTimeout(() => {
                            this.innerText = originalText;
                            this.classList.remove('copied');
                        }, 2000);
                    });
                });
            });
